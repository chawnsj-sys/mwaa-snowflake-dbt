FROM PUBLIC_MARTS.CUSTOMER_SUMMARY

销售趋势（按日）

SELECT order_date, COUNT(*) as order_count, SUM(amount) as revenue
FROM ANALYTICS.ORDERS
GROUP BY order_date ORDER BY order_date DESC LIMIT 30;

产品销售排名

SELECT p.product_name, COUNT(oi.order_id) as sold_count, SUM(oi.quantity * oi.unit_price) as revenue
FROM ANALYTICS.ORDER_ITEMS oi JOIN ANALYTICS.PRODUCTS p ON oi.product_id = p.product_id
GROUP BY 1 ORDER BY 3 DESC LIMIT 10;
